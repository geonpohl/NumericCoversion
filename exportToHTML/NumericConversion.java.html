import java.util.Scanner;
public class NumericConversion {

        int menuChoice = scnr.nextInt();

                System.out.print("Please enter an option: ");

                System.out.print("Please enter the numeric string to convert: ");
                short decimal = binaryStringDecode("");
                System.out.println("Result: " + decimal);
                System.out.println("Decoding Menu\n-------------\n1. Decode hexadecimal\n2. Decode binary\n3. Convert binary to hexadecimal\n4. Quit");
                System.out.print("Please enter an option: ");
                menuChoice = scnr.nextInt();

            } else if (menuChoice == 3) {
                System.out.print("Please enter the numeric string to convert: ");

                menuChoice = scnr.nextInt();

            } else {
                break;

    public static long hexStringDecode(String hex) {
        Scanner scnr = new Scanner(System.in);
        String hexaNum = scnr.nextLine();
        if(hexaNum.startsWith("0x")) {
            hexaNum = hexaNum.replaceAll("0x","");
            long decimal = Integer.parseInt(hexaNum,16);
            return decimal;
        }
        else {
            long decimal = Integer.parseInt(hexaNum, 16);
            return decimal;
        }
    }

    public static short binaryStringDecode(String binary) {
        Scanner scnr = new Scanner(System.in);
        String binaryNum = scnr.nextLine();
        if(binaryNum.startsWith("0b")) {
            binaryNum = binaryNum.replaceAll("0b","");
            int decimal = Integer.parseInt(binaryNum,2);
            return (short) decimal;
        } else  {
            int decimal = Integer.parseInt(binaryNum, 2);
            return (short) decimal;
        }
    }

    public static String binaryToHex(String binary) {
        Scanner scnr = new Scanner(System.in);
        String binaryNum = scnr.nextLine();
        int decimal = Integer.parseInt(binaryNum,2);
        String hexaNum = Integer.toHexString(decimal);
        return hexaNum;
    }

    public static short hexCharDecode(char digit) {

    }

}
